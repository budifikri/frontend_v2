Saat menampilkan log di Debug Log Viewer, lakukan parsing.

function formatJson(data) {
  if (!data) return "(empty)";

  if (typeof data === "string") {
    try {
      return JSON.stringify(JSON.parse(data), null, 2);
    } catch {
      return data;
    }
  }

  return JSON.stringify(data, null, 2);
}
